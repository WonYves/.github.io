# 观察者模式

观察者包含 观察目标和观察者两类对象 当一个对象状态发送改变 所有的依赖于它的对象都会得到通知并被自动更新 降低了目标者和观察者之间的耦合度 专注自身功能逻辑

 class sub{
        constructor(){
          this.observers = []
        }

        add(observer){
          this.observers.push(observer)
        }

        remove(observer){
          this.observers = this.observers.filter(item => item !== observer)
        }

        notify(){
          this.observers.forEach(item => {
            item.update()
          })
        }
    } 
    class observers{
      constructor(name){
        this.name = name
      }
      update(){
        console.log('update', this.name);
      }
    }

    // 实例化
    const subject = new sub()
    const observer1 = new observers('way')
    const observer2 = new observers('ray')

    // 添加观察者
    subject.add(observer1)
    subject.add(observer2)

    // 移除观察者
    subject.remove(observer1)
    
    // 遍历观察者并触发身上的方法
    setTimeout(() => {
      subject.notify()
    }, 1000)

class sub{
        constructor(){
          this.observers = []
        }

        add(observer){
          this.observers.push(observer)
        }

        remove(observer){
          this.observers = this.observers.filter(item => item !== observer)
        }

        notify(data){
          this.observers.forEach(item => {
            item.update(data)
          })
        }
    } 
    class observers{
      constructor(name){
        this.ele = document.querySelector(name)
      }
      update(data){
        this.ele.innerHTML = data
      }
    }

    // 实例化
    const subject = new sub()

    const observer1 = new observers('.bread')
    const observer2 = new observers('.header')
    subject.add(observer1)
    subject.add(observer2)

    const ubox = document.querySelectorAll('.ulbox li')
    for(let i=0 ;i<ubox.length;i++){
      ubox[i].onclick = function(){
        subject.notify(this.innerHTML)
      }
    }

# 发布订阅模式

观察者和目标者要相互知道 发布者和订阅者通过第三方调度 属于经过解耦合的观察者模式

const pubSub = {
      message: {},
      // 发布
      publish(type, cb) {        
        // 没有这个订阅者就直接返回
        if (!this.message[type]) return

        // 触发这个订阅者身上的订阅的所有方法
        this.message[type].forEach(item => item())
      },

      // 订阅
      subscribe(type, cb) {
        // 没有这个订阅者就先创建这个订阅者生成数组赋值传递过来的回调函数
        if (!this.message[type]) {
          this.message[type] = [cb]
        } else {
        // 有这个订阅者就往他身上继续添加传递进来的方法
          this.message[type].push(cb)
        }
      },

      // 取消订阅
      unscbscribe(type, cb) {
        // 没有这个订阅者就直接返回
        if (!this.message[type]) return
        // 没有传递订阅者对应的回调函数
        if (!cb) {
          // 清空订阅者
          this.message[type].length = 0
        }else{
          // 传了回调就过滤掉当前传递来的回调 取消订阅
          this.message[type] = this.message[type].filter(item => item != cb)
        }
        
      }

    }

    function testA() {
      console.log('testA');
    }
    function testB() {
      console.log('testB');
    }
    function testC() {
      console.log('testC');
    }
    function testD() {
      console.log('testD');
    }

    pubSub.subscribe('A', testA)
    pubSub.subscribe('A', testC)
    pubSub.subscribe('A', testD)
    pubSub.subscribe('B', testB)

    pubSub.unscbscribe('A', testC)

    pubSub.publish('A')
